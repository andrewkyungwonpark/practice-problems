function operation(a, b, op) { switch (op) { case 'add': return parseInt(a, 10) + parseInt(b, 10); break; case 'subtract': return parseInt(a, 10) - parseInt(b, 10); break; case 'multiply': return parseInt(a, 10) * parseInt(b, 10); break; case 'divide': return parseInt(a, 10) / parseInt(b, 10); break; } }
